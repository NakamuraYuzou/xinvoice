DROP VIEW 【XI】VBB経費パターン候補;

CREATE VIEW 【XI】VBB経費パターン候補 AS
 SELECT
	 B.取引先コード									 AS 取引先コード,
	 B.請求元										 AS 請求元,
	 B.経費タイプコード08							 AS 元経費タイプコード08,
	 B.経費タイプ名称08								 AS 元経費タイプ名称08,
	 B.経費タイプコード10							 AS 元経費タイプコード10,
	 B.経費タイプ名称10								 AS 元経費タイプ名称10,
	 A.【明細情報】明細項目							 AS 明細項目
 FROM 【XI】BBCSV A
	 INNER JOIN 【XI】BB請求元マスタ B ON A.【おもて情報】請求元 = B.請求元
	 LEFT JOIN 【XI】経費タイプパターン C ON B.取引先コード = C.取引先コード AND B.請求元 = C.請求元
									 AND A.【明細情報】明細項目 LIKE C.パターン
 WHERE C.取引先コード IS null
    AND  A.【明細情報】明細番号 != ''
 GROUP BY B.取引先コード , B.請求元 , B.経費タイプコード08 , B.経費タイプ名称08 ,
		  B.経費タイプコード10 , B.経費タイプ名称10 , A.【明細情報】明細項目
 ORDER BY A.【明細情報】明細項目;
